def binary_search(data, target):
    low = 0
    high = len(data) - 1

    while low <= high:
        mid = (low + high) // 2
        guess = data[mid]

        if guess == target:
            return mid  # Target found
        elif guess < target:
            low = mid + 1
        else:
            high = mid - 1

    return -1  # Target not found
